async function fetchLiveRates() { const status = document.getElementById('status'); const timeEl = document.getElementById('timestamp'); try { const response = await fetch("https://www.goldapi.io/api/XAU/INR", { method: 'GET', headers: { "x-access-token": MY_API_KEY, "Content-Type": "application/json" } }); if (!response.ok) throw new Error('API Sync Failed'); const data = await response.json(); // 1. Convert Global Ounce price to 1 Gram const globalPricePerGram = data.price / 31.1035; // 2. ADD INDIAN TAXES (Approx 9.18% for Duty + GST) // This aligns the global spot price with Mumbai retail rates. const mumbaiPricePerGram = globalPricePerGram * 1.0918; // 3. Calculate 10 Grams (24K) const base24k_10g = Math.round(mumbaiPricePerGram * 10); // 4. Calculate 22K (91.6% purity) and 18K (75% purity) const rate22k_10g = Math.round(base24k_10g * 0.916); const rate18k_10g = Math.round(base24k_10g * 0.75); // Update UI with Indian Number Formatting document.getElementById('p24').innerText = "₹" + base24k_10g.toLocaleString('en-IN'); document.getElementById('p22').innerText = "₹" + rate22k_10g.toLocaleString('en-IN'); document.getElementById('p18').innerText = "₹" + rate18k_10g.toLocaleString('en-IN'); const now = new Date(); timeEl.innerText = now.toLocaleTimeString(); status.innerText = "LIVE"; status.classList.add('live'); } catch (error) { console.error("Error:", error); status.innerText = "OFFLINE"; } }
